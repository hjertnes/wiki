This is how I run MSSQL on my work Mac. The TLDR is that I do C# development on a Mac.

version: "3"
services:
  sql-server:
    image: mcr.microsoft.com/mssql/server:2019-latest
    hostname: sql-server
    container_name: sql-server
    ports:
      - "1433:1433"
    environment:
      - ACCEPT_EULA=Y
      - SA_PASSWORD=MyPassword2020
      - MSSQL_PID=Express
